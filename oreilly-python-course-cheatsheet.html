3 · Lists & Loops

List Operations

fruits = ["apple", "banana", "cherry"]

# Modify
fruits.append("mango")      # add to end
fruits.insert(1, "kiwi")    # insert at index
fruits.remove("banana")     # remove by value
popped = fruits.pop()       # remove last
fruits.sort()               # sort in place
fruits.reverse()            # reverse in place

# Access
fruits[0]       # first element
fruits[-1]      # last element
fruits[1:3]     # slice
len(fruits)     # count

For Loops & List Comprehensions

# Basic for loop
for item in fruits:
    print(item)

# Loop with index
for i, item in enumerate(fruits):
    print(f"{i}: {item}")

# Loop over range
for i in range(5):     # 0,1,2,3,4
    print(i)

# List comprehension
squares = [x**2 for x in range(10)]
evens   = [x for x in range(20) if x % 2 == 0]
upper   = [s.upper() for s in fruits]

Chat Completions

    model="gpt-4o",           # or "gpt-4o-mini"
    messages=[
        {"role": "system",
         "content": "You are a helpful assistant."},
        {"role": "user",
         "content": "Explain recursion simply."}
    ]
)
text = response.choices[0].message.content
print(text)

Multi-Turn Conversation

messages = [
    {"role": "system", "content": "Be concise."}
]

def chat(user_msg):
    messages.append({"role":"user","content":user_msg})
    resp = client.chat.completions.create(
        model="gpt-4o-mini", messages=messages)
    reply = resp.choices[0].message.content
    messages.append({"role":"assistant","content":reply})
    return reply

13 · Prompt Engineering

Prompt Templates

def summarize(text, max_words=100):
    prompt = f"""
Summarize the following text in {max_words} words.
Focus on key points only.

TEXT:
{text}

SUMMARY:"""
    return ask_ai(prompt)

def categorize(item, categories):
    cats = ", ".join(categories)
    prompt = f"""
Categorize this item into one of: {cats}
Item: {item}
Return only the category name."""
    return ask_ai(prompt)

Prompt Chaining

# Step 1: Generate idea
idea = ask_ai("Suggest a creative image prompt")

# Step 2: Critique it
critique = ask_ai(
    f"Critique this prompt and suggest improvements:\n{idea}"
)

# Step 3: Refine
refined = ask_ai(
    f"Original: {idea}\nFeedback: {critique}\n"
    f"Write an improved version:"
)

# Step 4: Generate image
img = client.images.generate(
    model="dall-e-3", prompt=refined)

Structured Output Prompt

prompt = """
Extract invoice data as JSON with this schema:
{
  "vendor": "string",
  "total": number,
  "date": "YYYY-MM-DD",
  "items": [{"desc": "string", "price": number}]
}

Invoice text:
{invoice_text}

Return only valid JSON, no markdown."""
Tip: Always tell the model the exact output format you need. Say "Return only JSON, no explanation" to avoid markdown code fences you'd have to strip.

14 · Automation Project Patterns

File Organizer

import os
import shutil
from pathlib import Path

folder = Path("./downloads")
rules = {
    ".pdf": "docs", ".jpg": "images",
    ".png": "images", ".csv": "data"
}

for file in folder.iterdir():
    if file.is_file():
        dest = rules.get(file.suffix, "other")
        dest_path = folder / dest
        dest_path.mkdir(exist_ok=True)
        shutil.move(str(file), dest_path / file.name)

PDF Summarizer

import PyPDF2

def summarize_pdf(path):
    with open(path, "rb") as f:
        reader = PyPDF2.PdfReader(f)
        text = ""
        for page in reader.pages:
            text += page.extract_text()

    return ask_ai(f"Summarize this paper:\n{text[:4000]}")

summary = summarize_pdf("paper.pdf")
print(summary)
